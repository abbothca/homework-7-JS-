Я написала два алгоритми. Метод 1 повертає новий сортований масив. Метод 2 - може посортувати вхідний масив, не треба створювати інший. Але для методу 2 - я не можу присвоїти замість null і undefined - [] (єдиний варіант - теж створити копію, і повертати або відсортовану копію або []). 

          
            function sortArray(array) {
    

              if (array) {
          
                  array.forEach( (value, index, array) => {
                  
                      let indexOfCurrentElement = index;
                      let currentValue = value; 
                      while (indexOfCurrentElement > 0 && array[indexOfCurrentElement - 1] > value) {
                          
                          array[indexOfCurrentElement] = array[indexOfCurrentElement - 1];
                          indexOfCurrentElement--;
                      }
                      array[indexOfCurrentElement] = currentValue;
                  });
          
                  return array;
              } else {
                array = [];  //         !!!!!не працює!!!!!!
              }   
          };
